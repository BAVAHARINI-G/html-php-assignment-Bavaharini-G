<?php
// Sample string containing multiple email addresses
$text = "Please contact john.doe@example.com, support@my-site.org, 
or admin123@company.net for more details.";

// Regular expression to match valid email addresses
$pattern = '/[a-zA-Z0-9._%+-]+@[a-zA-Z0-9.-]+\.[a-z]{2,}/';

// Use preg_match_all to find all matches
preg_match_all($pattern, $text, $matches);

// $matches[0] contains the array of found email addresses
$emails = $matches[0];

// Display the emails
echo "Extracted Email Addresses:<br>";
foreach ($emails as $email) {
    echo $email . "<br>";
}
?>

OUTPUT:

Extracted Email Addresses:
john.doe@example.com
support@my-site.org
admin123@company.net
