<?php
// Initial array
$x = array(1, 2, 3, 4, 5);

// Delete element (for example, remove value 3)
unset($x[2]); // Index 2 contains the value 3

// Re-index the array to avoid gaps
$x = array_values($x);

// Print array elements
foreach ($x as $value) {
    echo $value . " ";
}
?>


OUTPUT:

1 2 4 5
